country_event = {
	id = operations.1
	is_triggered_only = yes
	hidden = yes
	option = {
		log = "Clearing array:"
		845 = {
			log = "[THIS.GetName]"
			clear_array = provinces_@THIS
		}
	}
}

		}
	}

#Debug event to check what provinces are in a state
country_event = {
	id = operations.4
	is_triggered_only = yes
	hidden = yes
	option = {
		for_each_scope_loop = {
			array = states_required_checking

			log = "Provinces in State: [?THIS.GetName]"

			for_each_loop = {
				array = provinces_@THIS

				log = "Province ID: [?v]"
			}
		}
	}
}
